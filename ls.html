ls コマンド

タイムスタンプを YYYY-MM-DD hh:mm:ss のフォーマットで表示するには 2013/04/03

ls -alF --time-style="+%Y-%m-%d %H:%M:%S"

find の場合

find . -printf "%M %n %u %g %k %TY-%Tm-%Td %TH:%TM:%TS %p\n"

ls -alF --time-style="+%Y-%m-%d %H:%M:%S" | sort -r -k6,7
