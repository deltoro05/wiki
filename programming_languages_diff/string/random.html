ランダム文字列を生成するには

PHP 2013/07/24

サンプル

function createpasswd(){
  $data = 'abcdefghkmnprstuvwxyzABCDEFGHJKLMNPRSTUVWXYZ2345678';
  $n = strlen($data);
  $ret = '';
  for($ii=0;$ii<8;$ii++){
    $ret .= $data[rand(0, $n-1)];
  }
  return $ret;
}

function createpasswd_num(){
  $data = '1234567890';
  $n = strlen($data);
  $ret = '';
  for($ii=0;$ii<8;$ii++){
    $ret .= $data[rand(0, $n-1)];
  }
  return $ret;
}
