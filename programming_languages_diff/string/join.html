Perl 2013/07/27

my @arr = ('abc', 'def', 'ghi');
print @arr;            # => abcdefghi
print "@arr";          # => abc def ghi
print join(' ', @arr); # => abc def ghi
print join(',', @arr); # => abc,def,ghi
