function recursive($dir) {
  if (!file_exists($dir)) {
    return;
  }
  $dhandle = opendir($dir);
  if ($dhandle) {
    while (false !== ($fname = readdir($dhandle))) {
       if (is_dir( "{$dir}/{$fname}" )) {
          if (($fname != '.') && ($fname != '..')) {
             recursive("$dir/$fname");
          }
       } else {
          foo($dir/$fname);
       }
    }
    closedir($dhandle);
  }
  foo($dir);
}

Perl 2015/04/08

sub recursive {
    my ($dir, $p) = @_;
    my @files = glob("$dir/*");
    foreach my $file (@files) {
        if (-d $file) {
            recursive($file, $p);
        } else {
            $p->($file);
        }
    }
}
