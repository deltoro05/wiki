三角関数・指数関数

Java / Scala 2014/07/11

Signature:

// java.lang.Math

static double PI // 円周率
static double E  // 自然対数の底

static double abs(int x)    // 絶対値
static double abs(long x)   // 絶対値
static double abs(float x)  // 絶対値
static double abs(double x) // 絶対値

static double sqrt(double x) // 平方根
static double cbrt(double x) // 立方根

static double exp(double x) // 指数関数

static double log(double x)   // 対数
static double log10(double x) // 対数

static double cos(double x) // 三角関数
static double sin(double x) // 三角関数
static double tan(double x) // 三角関数

static double acos(double x)            // 逆三角関数
static double asin(double x)            // 逆三角関数
static double atan(double x)            // 逆三角関数
static double atan2(double x, double y) // 逆三角関数

static double cosh(double x) // 双曲線関数
static double sinh(double x) // 双曲線関数
static double tanh(double x) // 双曲線関数

PHP 2014/07/11

$z = M_PI; // 円周率
$z = M_E;  // 自然対数の底

$z = abs($x); // 絶対値

$z = sqrt($x); // 平方根

$z = exp($x); // 指数関数

$z = log($x);   // 対数
$z = log10($x); // 対数

$z = cos($x); // 三角関数
$z = sin($x); // 三角関数
$z = tan($x); // 三角関数

$z = acos($x);      // 逆三角関数
$z = asin($x);      // 逆三角関数
$z = atan($x);      // 逆三角関数
$z = atan2($x, $y); // 逆三角関数

$z = cosh($x); // 双曲線関数
$z = sinh($x); // 双曲線関数
$z = tanh($x); // 双曲線関数

Perl 2014/08/02

# 絶対値
$z = abs($x));  # `use POSIX` が要

# 平方根
$z = sqrt($x);

# 指数関数
$z = exp($x);

# 対数
$z = log($x);
$z = log10($x)); # `use POSIX` が要

# 三角関数
$z = sin($x);
$z = cos($x);
$z = tan($x));  # `use POSIX` が要

# 逆三角関数
$z = asin($x)); # `use POSIX` が要
$z = acos($x)); # `use POSIX` が要
$z = atan2($x, $y);

# 双曲線関数
$z = cosh($x)); # `use POSIX` が要
$z = sinh($x)); # `use POSIX` が要
$z = tanh($x)); # `use POSIX` が要

いくつかの関数は use POSIX; と宣言しないと使えない。
