Ruby / JRuby 2014/12/22

Hash#delete(key)

例

hash = {'a' => 10, 'b' => 20, 'c' => 30}

p hash
# => {"a"=>10, "b"=>20, "c"=>30}

hash.delete('b')
p hash
# => {"a"=>10, "c"=>30}

Perl 2014/12/22

deleteという関数を利用する。

ハッシュから要素を削除する例

my %data = ('a' => 10, 'b' => 20, 'c' => 30);
print Dumper \%data;
# 出力結果
# $VAR1 = {
#           'c' => 30,
#           'a' => 10,
#           'b' => 20
#         };

delete($data{'b'});
print Dumper \%data;
# 出力結果
# $VAR1 = {
#           'c' => 30,
#           'a' => 10
#         };

ハッシュへのレファレンスから要素を削除する例

my $data = {'a' => 10, 'b' => 20, 'c' => 30};
print Dumper $data;
# 出力結果
# $VAR1 = {
#           'c' => 30,
#           'a' => 10,
#           'b' => 20
#         };

delete($data->{'b'});
print Dumper $data;
# 出力結果
# $VAR1 = {
#           'c' => 30,
#           'a' => 10
#         };
